  - Demo 2: End‑to‑End File Analysis                                                                                                                                                                                                                                                                           
      - Windows PowerShell                                                                                                                                                                                                                                                                                     
          - $res = Invoke-RestMethod -Method Post -Uri "http://localhost:8080/analyze" -Form @{ file = Get-Item ".\sample.bin" }                                                                                                                                                                               
          - $job = $res.job_id; echo "JOB: $job"                                                                                                                                                                                                                                                               
          - while ($true) { $st = Invoke-RestMethod "http://localhost:8080/result/$job"; $st.state; if ($st.state -in "done","failed") { $st | ConvertTo-Json -Depth 6; break }; Start-Sleep -Seconds 1 }                                                                                                      
          - If json_url exists:                                                                                                                                                                                                                                                                                
              - $jurl = $st.export.json_url                                                                                                                                                                                                                                                                    
              - Invoke-RestMethod -Uri ("http://localhost:8090" + $jurl) | ConvertTo-Json -Depth 6                                                                                                                                                                                                             
      - Bash                                                                                                                                                                                                                                                                                                   
          - JOB=$(curl -s -F "file=@./sample.bin" http://localhost:8080/analyze | jq -r .job_id); echo "JOB=$JOB"
          - while true; do ST=$(curl -s "http://localhost:8080/result/$JOB"); echo "$ST" | jq -r .state; [[ "$(echo "$ST" | jq -r .state)" =~ ^(done|failed)$ ]] && echo "$ST" | jq . && break; sleep 1; done                                                                                                  
          - JURL=$(echo "$ST" | jq -r '.export.json_url // empty'); [ -n "$JURL" ] && curl -s "http://localhost:8090$JURL" | jq .

  - Demo 4: Encrypted Archive Flow (ZIP password)                                                                                                                                                                                                                                                              
      - Submit the provided enc.zip (if it is passworded)                                                                                                                                                                                                                                                      
          - PowerShell:                                                                                                                                                                                                                                                                                        
              - $res = Invoke-RestMethod -Method Post -Uri "http://localhost:8080/analyze" -Form @{ file = Get-Item ".\enc.zip" }                                                                                                                                                                              
              - $job = $res.job_id
